```python
import os, re, sys

SRC = "src"
link_re = re.compile(r'\]\(\s*(<[^>]+>|[^)]+)\)')

# 알려진 pre-existing false-positive (파일 상대경로, 링크 타깃 원문) — DEAD 집계 제외
FALSE_POS = {
    ("src/introduction.md", "images/intro/파일명.jpg"),
    ("src/template.md", "session-N"),
}

total = 0
raw_dead = []   # 모든 미존재 타깃
for root, _, files in os.walk(SRC):
    for fn in sorted(files):
        if not fn.endswith(".md"):
            continue
        path = os.path.join(root, fn)
        rel = os.path.relpath(path)
        with open(path, encoding="utf-8") as f:
            text = f.read()
        for m in link_re.finditer(text):
            target = m.group(1).strip()
            if target.startswith("<") and target.endswith(">"):
                target = target[1:-1].strip()
            if target.startswith(("http://", "https://", "mailto:", "tel:", "#")):
                continue
            noanchor = target.split("#", 1)[0].strip()
            if not noanchor:
                continue
            total += 1
            resolved = os.path.normpath(os.path.join(os.path.dirname(path), noanchor))
            if not os.path.exists(resolved):
                raw_dead.append((rel, target, resolved))

# false-positive 필터
def is_fp(rel, target):
    noanchor = target.split("#", 1)[0].strip()
    return (rel, target) in FALSE_POS or (rel, noanchor) in FALSE_POS

filtered_dead = [d for d in raw_dead if not is_fp(d[0], d[1])]

print("TOTAL internal links:", total)
print("RAW DEAD (incl. known false-positives):", len(raw_dead))
for d in raw_dead:
    tag = " [known-FP]" if is_fp(d[0], d[1]) else ""
    print(f"  {d[0]} -> {d[1]}  ({d[2]}){tag}")
print("DEAD (filtered):", len(filtered_dead))
sys.exit(1 if filtered_dead else 0)
```
